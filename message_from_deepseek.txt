# Добавляем Малышу новый метод
def say_hello(self):
    print("Привет, капитан! Малыш получил новую способность.")

# Привязываем метод к экземпляру malysh
malysh.say_hello = say_hello.__get__(malysh)

# Вызываем новый метод
malysh.say_hello()
def multiply(self, a, b):
    print(f"Произведение {a} и {b} равно {a*b}")
malysh.multiply = multiply.__get__(malysh)
malysh.multiply(4, 5)
